¿Qué hace Frigo Sec AM?

Un sistema completo basado en Arduino que cubre control de acceso, indicación visual y sonora, y respuesta ante emergencias.

🔑

Acceso por Código

Ingreso mediante teclado matricial 4×4 con código de 5 dígitos configurable. Solo el personal autorizado puede acceder.

🚨

Emergencia Interna

Botón físico dentro de la cámara que abre la puerta de inmediato, activando alarma continua para alertar al exterior.

⚙️

Servo Motor

Control mecánico preciso de la cerradura. Gira 90° para abrir y vuelve a 0° al cerrar. Fuerza y velocidad configurables.

💡

LEDs Indicadores

LED verde para acceso permitido, LED rojo para acceso denegado. Retroalimentación visual instantánea sin ambigüedad.

🔊

Alertas Sonoras

Buzzer con tonos diferenciados: confirmación de acceso (1000 Hz), error de código (400 Hz), y alarma de emergencia continua.

🛡️

Código Configurable

La clave maestra se actualiza directamente en el código fuente. Permite rotación de credenciales según protocolos del establecimiento.

Lógica del sistema

Flujo de funcionamiento

Dos modos de operación: acceso normal por código y emergencia de apertura inmediata.

01

Ingreso de código

El operario presiona hasta 5 teclas en el teclado 4×4. Cada pulsación se registra internamente.

02

Confirmación con #

Al presionar # el Arduino compara la entrada con la clave maestra almacenada en memoria.

03A

Acceso concedido ✅

Servo → 90°, LED verde enciende, buzzer emite tono corto. Puerta permanece abierta 3 segundos.

03B

Acceso denegado ❌

LED rojo enciende, buzzer emite alerta de error 400 Hz. Sistema se reinicia para nuevo intento.

🆘
Protocolo de Emergencia Interna
→ Sensor analógico A0 detecta pulsación (valor > 500)
→ Servo gira inmediatamente a 90° — puerta abierta
→ LED verde enciende indicando salida libre
→ Buzzer activa alarma continua a 1000 Hz
→ Sistema mantiene estado durante 5 segundos
→ Buzzer se apaga, servo vuelve a 0° — reinicio completo

Código C++ / Arduino

seguridad_frigorifico.ino — Frigo Sec AM 
// ═══════════════════════════════════════════════════════
//  Frigo Sec AM · Sistema de Seguridad para Frigorífico
//  Alumnas: Santillan Morena y Alegre Ariana
//  Curso 7°2 · Prof. Coronel Leonel · 22/04/2026
// ═══════════════════════════════════════════════════════

#include <Keypad.h>
#include <Servo.h>

// ── TECLADO ───────────────────────────────────────────
const byte FILAS = 4, COLUMNAS = 4;

char teclas[FILAS][COLUMNAS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte pinesFilas[FILAS]       = {9, 8, 7, 6};
byte pinesColumnas[COLUMNAS] = {5, 4, 3, 2};

Keypad teclado = Keypad(
  makeKeymap(teclas), pinesFilas, pinesColumnas, FILAS, COLUMNAS
);

// ── PINES ─────────────────────────────────────────────
Servo miServo;
const int PIN_ROJO      = 11;
const int PIN_VERDE     = 12;
const int PIN_BUZZER    = 13;
const int PIN_EMERGENCIA = A0;

// ── ACCESO ────────────────────────────────────────────
char claveMaestra[] = "12345";
char claveIngresada[6];
byte indice = 0;

void setup() {
  miServo.attach(10);
  miServo.write(0);  // Cerradura trabada
  pinMode(PIN_ROJO,       OUTPUT);
  pinMode(PIN_VERDE,      OUTPUT);
  pinMode(PIN_BUZZER,     OUTPUT);
  pinMode(PIN_EMERGENCIA, INPUT);
  digitalWrite(PIN_ROJO,  LOW);
  digitalWrite(PIN_VERDE, LOW);
}

void loop() {

  // 🚨 EMERGENCIA — prioridad máxima ────────────────────
  if (analogRead(PIN_EMERGENCIA) > 500) {
    digitalWrite(PIN_VERDE, HIGH);
    tone(PIN_BUZZER, 1000);   // Alarma continua
    miServo.write(90);         // Abre puerta
    delay(5000);
    noTone(PIN_BUZZER);
    miServo.write(0);
    digitalWrite(PIN_VERDE, LOW);
    return;
  }

  // 🔐 ACCESO NORMAL ────────────────────────────────────
  char tecla = teclado.getKey();
  if (!tecla) return;

  if (tecla == '*') {          // Borrar
    indice = 0;
    memset(claveIngresada, 0, sizeof(claveIngresada));
    return;
  }

  if (tecla != '#' && indice < 5) {
    claveIngresada[indice++] = tecla;
    return;
  }

  if (tecla == '#') {
    claveIngresada[indice] = '\0';

    if (strcmp(claveIngresada, claveMaestra) == 0) {
      // ✅ Correcto
      digitalWrite(PIN_VERDE, HIGH);
      tone(PIN_BUZZER, 1000); delay(200); noTone(PIN_BUZZER);
      miServo.write(90);
      delay(3000);
      miServo.write(0);
      digitalWrite(PIN_VERDE, LOW);
    } else {
      // ❌ Incorrecto
      digitalWrite(PIN_ROJO, HIGH);
      tone(PIN_BUZZER, 400); delay(500); noTone(PIN_BUZZER);
      delay(1000);
      digitalWrite(PIN_ROJO, LOW);
    }

    indice = 0;
    memset(claveIngresada, 0, sizeof(claveIngresada));
  }
}
